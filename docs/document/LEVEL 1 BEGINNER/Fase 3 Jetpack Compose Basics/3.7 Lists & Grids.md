### ğŸ“ Penjelasan Konsep:
**LazyColumn seperti Eskalator:**
- Hanya item yang terlihat yang dirender
- Hemat memory dan performa
- Gunakan `items()`, `itemsIndexed()`, dan `item()` untuk berbagai kebutuhan

**Fungsi Penting:**
- `items()`: Untuk list biasa
- `itemsIndexed()`: Dapatkan index item
- `item()`: Untuk item single (header/footer)
- `key`: Identifikasi unik untuk item
- `contentType`: Kategori tipe konten untuk optimasi
- `contentPadding`: Padding dalam list
- `reverseLayout`: Urutan terbalik

@Composable
fun LoadMoreExamplesScreen() {
    var items by remember { mutableStateOf(emptyList<String>()) }
    var isLoading by remember { mutableStateOf(false) }
    var hasMore by remember { mutableStateOf(true) }
    val lazyListState = rememberLazyListState()
    
    Column(
        modifier = Modifier
            .fillMaxSize()
            .padding(16.dp),
        verticalArrangement = Arrangement.spacedBy(16.dp)
    ) {
        Text("8ï¸âƒ£ Load More / Pagination", fontSize = 20.sp, fontWeight = FontWeight.Bold)
        
        // 1. Basic Load More Implementation
        Text("Basic Load More Implementation", fontWeight = FontWeight.Bold)
        
        LazyColumn(
            state = lazyListState,
            modifier = Modifier
                .fillMaxWidth()
                .height(350.dp)
                .background(Color(0xFFE3F2FD)),
            contentPadding = PaddingValues(8.dp)
        ) {
            // Load initial data if empty
            if (items.isEmpty() && !isLoading) {
                items(1) {
                    LaunchedEffect(Unit) {
                        loadMoreData { newItems, hasMoreResult ->
                            items = newItems
                            hasMore = hasMoreResult
                            isLoading = false
                        }
                    }
                }
            }
            
            items(items) { item ->
                Card(
                    modifier = Modifier
                        .fillMaxWidth()
                        .padding(vertical = 4.dp)
                ) {
                    Text(
                        text = item,
                        modifier = Modifier.padding(16.dp)
                    )
                }
            }
            
            // Loading indicator
            if (isLoading) {
                item {
                    Box(
                        modifier = Modifier
                            .fillMaxWidth()
                            .padding(16.dp),
                        contentAlignment = Alignment.Center
                    ) {
                        CircularProgressIndicator()
                    }
                }
            }
            
            // Load more item (triggered when scrolled to bottom)
            if (hasMore && !isLoading) {
                item {
                    Box(
                        modifier = Modifier
                            .fillMaxWidth()
                            .padding(16.dp),
                        contentAlignment = Alignment.Center
                    ) {
                        Button(
                            onClick = {
                                if (!isLoading) {
                                    isLoading = true
                                    loadMoreData { newItems, hasMoreResult ->
                                        items = items + newItems
                                        hasMore = hasMoreResult
                                        isLoading = false
                                    }
                                }
                            },
                            enabled = !isLoading
                        ) {
                            if (isLoading) {
                                Text("Memuat...")
                            } else {
                                Text("Muat Lebih Banyak")
                            }
                        }
                    }
                }
            }
        }
        
        // Auto-load more when reaching end
        LaunchedEffect(lazyListState) {
            snapshotFlow { lazyListState.layoutInfo.visibleItemsInfo.lastOrNull()?.index }
                .collect { lastVisibleIndex ->
                    val totalItems = items.size
                    if (lastVisibleIndex != null && lastVisibleIndex >= totalItems - 3 && hasMore && !isLoading) {
                        isLoading = true
                        loadMoreData { newItems, hasMoreResult ->
                            items = items + newItems
                            hasMore = hasMoreResult
                            isLoading = false
                        }
                    }
                }
        }
        
        // Control buttons
        Row(
            modifier = Modifier.fillMaxWidth(),
            horizontalArrangement = Arrangement.spacedBy(8.dp)
        ) {
            Button(
                onClick = {
                    items = emptyList()
                    hasMore = true
                    isLoading = false
                }
            ) {
                Text("Reset")
            }
            
            Button(
                onClick = {
                    if (!isLoading && hasMore) {
                        isLoading = true
                        loadMoreData { newItems, hasMoreResult ->
                            items = items + newItems
                            hasMore = hasMoreResult
                            isLoading = false
                        }
                    }
                },
                enabled = hasMore && !isLoading
            ) {
                Text("Manual Load")
            }
        }
        
        Text("Status: Items = ${items.size}, Has More = $hasMore, Loading = $isLoading")
    }
}
