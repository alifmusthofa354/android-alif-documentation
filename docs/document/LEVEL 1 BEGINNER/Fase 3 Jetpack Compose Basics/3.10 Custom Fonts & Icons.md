### ğŸ“ Penjelasan Konsep:

**Langkah Setup:**

@Composable
fun FontWeightsDemo(fontFamily: FontFamily) {
    val weights = listOf(
        FontWeight.Thin to "Thin",
        FontWeight.ExtraLight to "ExtraLight",
        FontWeight.Light to "Light",
        FontWeight.Normal to "Normal",
        FontWeight.Medium to "Medium",
        FontWeight.SemiBold to "SemiBold",
        FontWeight.Bold to "Bold",
        FontWeight.ExtraBold to "ExtraBold",
        FontWeight.Black to "Black"
    )
    
    Column(
        verticalArrangement = Arrangement.spacedBy(8.dp)
    ) {
        weights.forEach { (weight, name) ->
            Card(
                modifier = Modifier.fillMaxWidth()
            ) {
                Row(
                    modifier = Modifier
                        .fillMaxWidth()
                        .padding(12.dp),
                    verticalAlignment = Alignment.CenterVertically
                ) {
                    Text(
                        text = "$name:",
                        fontWeight = FontWeight.Medium,
                        modifier = Modifier.width(100.dp)
                    )
                    Text(
                        text = "The quick brown fox jumps over the lazy dog",
                        fontFamily = fontFamily,
                        fontWeight = weight,
                        fontSize = 14.sp,
                        maxLines = 1,
                        overflow = TextOverflow.Ellipsis
                    )
                }
            }
        }
    }
}

- `Icon` dengan `painter`: Menggunakan custom vector icon

7. Atur size dan tint sesuai konteks penggunaan
