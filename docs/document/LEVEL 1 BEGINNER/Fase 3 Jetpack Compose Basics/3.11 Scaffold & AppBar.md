package com.example.mycomposeapp.ui.screens

import androidx.compose.foundation.layout.*
import androidx.compose.foundation.lazy.LazyColumn
import androidx.compose.material.icons.Icons
import androidx.compose.material.icons.filled.*
import androidx.compose.material3.*
import androidx.compose.runtime.*
import androidx.compose.ui.Alignment
import androidx.compose.ui.Modifier
import androidx.compose.ui.text.font.FontWeight
import androidx.compose.ui.unit.dp
import androidx.compose.ui.unit.sp

@Composable
fun ScaffoldBasicsScreen() {
    val scaffoldState = rememberScaffoldState()
    
    Scaffold(
        scaffoldState = scaffoldState,
        topBar = {
            TopAppBar(
                title = { Text("Scaffold Basics") },
                navigationIcon = {
                    IconButton(onClick = { /* Handle navigation */ }) {
                        Icon(Icons.Default.Menu, contentDescription = "Menu")
                    }
                },
                actions = {
                    IconButton(onClick = { /* Handle search */ }) {
                        Icon(Icons.Default.Search, contentDescription = "Search")
                    }
                }
            )
        },
        bottomBar = {
            BottomAppBar {
                IconButton(onClick = { /* Handle home */ }) {
                    Icon(Icons.Default.Home, contentDescription = "Home")
                }
                IconButton(onClick = { /* Handle favorite */ }) {
                    Icon(Icons.Default.Favorite, contentDescription = "Favorite")
                }
                IconButton(onClick = { /* Handle profile */ }) {
                    Icon(Icons.Default.Person, contentDescription = "Profile")
                }
            }
        },
        floatingActionButton = {
            ExtendedFloatingActionButton(
                text = { Text("Tambah") },
                icon = { Icon(Icons.Default.Add, contentDescription = "Tambah") },
                onClick = { 
                    // Handle FAB click
                    println("FAB clicked!")
                }
            )
        },
        content = { padding ->
            LazyColumn(
                modifier = Modifier
                    .fillMaxSize()
                    .padding(padding),
                contentPadding = PaddingValues(16.dp),
                verticalArrangement = Arrangement.spacedBy(8.dp)
            ) {
                items(20) { index ->
                    Card(
                        modifier = Modifier
                            .fillMaxWidth()
                    ) {
                        Text(
                            text = "Item ${index + 1}",
                            modifier = Modifier.padding(16.dp),
                            fontSize = 16.sp
                        )
                    }
                }
            }
        }
    )
}
```

package com.example.mycomposeapp.ui.screens

import androidx.compose.ui.unit.sp

@Composable
fun TopAppBarSmallScreen() {
    val scaffoldState = rememberScaffoldState()
    
    Scaffold(
        scaffoldState = scaffoldState,
        topBar = {
            SmallTopAppBar(
                title = { Text("Small TopAppBar") },
                navigationIcon = {
                    IconButton(onClick = { /* Handle back */ }) {
                        Icon(Icons.AutoMirrored.Filled.ArrowBack, contentDescription = "Kembali")
                    }
                },
                actions = {
                    IconButton(onClick = { /* Handle search */ }) {
                        Icon(Icons.Default.Search, contentDescription = "Cari")
                    }
                    IconButton(onClick = { /* Handle more */ }) {
                        Icon(Icons.Default.MoreVert, contentDescription = "Lainnya")
                    }
                },
                colors = TopAppBarDefaults.topAppBarColors(
                    containerColor = MaterialTheme.colorScheme.primary,
                    titleContentColor = MaterialTheme.colorScheme.onPrimary,
                    navigationIconContentColor = MaterialTheme.colorScheme.onPrimary,
                    actionIconContentColor = MaterialTheme.colorScheme.onPrimary
                )
            )
        },
        content = { padding ->
            LazyColumn(
                modifier = Modifier
                    .fillMaxSize()
                    .padding(padding)
            ) {
                item {
                    // Example content
                    Card(
                        modifier = Modifier
                            .fillMaxWidth()
                            .padding(16.dp)
                    ) {
                        Text(
                            text = "Konten yang tidak tertutup oleh AppBar",
                            modifier = Modifier.padding(16.dp),
                            fontSize = 16.sp
                        )
                    }
                }
                
                items(15) { index ->
                    ListTile(
                        leading = {
                            Icon(
                                imageVector = when (index % 3) {
                                    0 -> Icons.Default.Email
                                    1 -> Icons.Default.Phone
                                    else -> Icons.Default.Person
                                },
                                contentDescription = null
                            )
                        },
                        headlineContent = { Text("Kontak ${index + 1}") },
                        supportingContent = { Text("Deskripsi kontak $index") },
                        modifier = Modifier.clickable { /* Handle click */ }
                    )
                }
            }
        }
    )
}

@Composable
fun ListTile(
    leading: @Composable () -> Unit,
    headlineContent: @Composable () -> Unit,
    supportingContent: @Composable () -> Unit,
    modifier: Modifier = Modifier
) {
    Row(
        modifier = modifier
            .fillMaxWidth()
            .padding(horizontal = 16.dp, vertical = 12.dp),
        verticalAlignment = Alignment.CenterVertically
    ) {
        leading()
        Spacer(modifier = Modifier.width(16.dp))
        Column(
            modifier = Modifier.weight(1f)
        ) {
            headlineContent()
            supportingContent()
        }
    }
}
```

- `actions`: Ikon aksi (biasanya di sisi kanan)
- `colors`: Styling untuk berbagai elemen

**Fungsi Utama:**
- Navigasi: Back button, menu
- Aksi: Search, filter, share
- Judul: Nama layar/section
- Status: Indikator atau info

        modifier = Modifier.fillMaxSize()

                .fillMaxWidth()

                type = ScrollBehaviorType.Pinned,
                currentType = scrollBehaviorType,
                onClick = { scrollBehaviorType = it },
                text = "Pinned"
            )
            
            ScrollBehaviorButton(
                type = ScrollBehaviorType.EnterAlways,
                currentType = scrollBehaviorType,
                onClick = { scrollBehaviorType = it },
                text = "Enter Always"
            )
            
            ScrollBehaviorButton(
                type = ScrollBehaviorType.ExitUntilCollapsed,
                currentType = scrollBehaviorType,
                onClick = { scrollBehaviorType = it },
                text = "Exit Until"
            )
        }
        
        // Content with different scroll behaviors
        when (scrollBehaviorType) {
            ScrollBehaviorType.Pinned -> {
                val scrollBehavior = TopAppBarDefaults.pinnedScrollBehavior()
                
                Scaffold(
                    topBar = {
                        LargeTopAppBar(
                            title = { Text("Pinned AppBar") },
                            navigationIcon = {
                                IconButton(onClick = { /* Handle navigation */ }) {
                                    Icon(Icons.Default.Menu, contentDescription = "Menu")

                        )
                    },

                        LazyColumn(
                            modifier = Modifier
                                .fillMaxSize()
                                .padding(padding)
                                .nestedScroll(scrollBehavior.nestedScrollConnection),
                            contentPadding = PaddingValues(16.dp)
                        ) {
                            items(30) { index ->
                                Card(
                                    modifier = Modifier
                                        .fillMaxWidth()
                                        .padding(vertical = 4.dp)
                                ) {
                                    Text(
                                        text = "Item ${index + 1} - " + 
                                               "Pinned AppBar tidak pernah menghilang saat scroll",
                                        modifier = Modifier.padding(16.dp)
                                    )
                                }
                            }
                        }
                    }
                )
            }
            ScrollBehaviorType.EnterAlways -> {
                val scrollBehavior = TopAppBarDefaults.enterAlwaysScrollBehavior()
                
                Scaffold(
                    topBar = {
                        LargeTopAppBar(
                            title = { Text("Enter Always AppBar") },
                            navigationIcon = {
                                IconButton(onClick = { /* Handle navigation */ }) {
                                    Icon(Icons.Default.Menu, contentDescription = "Menu")
                                }
                            },
                            actions = {
                                IconButton(onClick = { /* Handle search */ }) {
                                    Icon(Icons.Default.Search, contentDescription = "Search")
                                }
                            },
                            scrollBehavior = scrollBehavior
                        )
                    },
                    content = { padding ->
                        LazyColumn(
                            modifier = Modifier
                                .fillMaxSize()
                                .padding(padding)
                                .nestedScroll(scrollBehavior.nestedScrollConnection),
                            contentPadding = PaddingValues(16.dp)
                        ) {

                                Card(
                                    modifier = Modifier
                                        .fillMaxWidth()
                                        .padding(vertical = 4.dp)

                                        text = "Item ${index + 1} - " +

                Scaffold(
                    topBar = {
                        LargeTopAppBar(

    val scaffoldState = rememberScaffoldState()
    var searchExpanded by remember { mutableStateOf(false) }

                        text = if (searchQuery.isBlank()) {
                            "Total: ${searchableItems.size} item"
                        } else {
                            "Ditemukan: ${filteredItems.size} item"
                        },
                        fontSize = 14.sp,
                        color = MaterialTheme.colorScheme.onSurfaceVariant
                    )
                }
                
                items(filteredItems) { item ->
                    Card(
                        modifier = Modifier.fillMaxWidth()

## ðŸŽ¯ Kesimpulan

Dengan menguasai Scaffold & AppBar di Jetpack Compose, kamu sekarang memiliki kemampuan untuk:

- **Menggunakan Scaffold**: Membangun struktur aplikasi yang konsisten
- **Mengelola TopAppBar**: Menggunakan berbagai jenis app bar (Small, Medium, Large, CenterAligned)
- **Mengimplementasikan Scroll Behavior**: Mengatur perilaku app bar saat scroll
- **Menggunakan BottomAppBar**: Menyediakan navigasi di bagian bawah
- **Menggunakan FloatingActionButton**: Menerapkan berbagai jenis FAB
- **Mengelola Scaffold Slots**: Menggunakan semua slot dalam struktur Scaffold
- **Membuat Layar Lengkap**: Menggabungkan semua komponen dalam satu layar
- **Mengimplementasikan Search**: Menambahkan fitur pencarian di app bar

### ðŸ”¥ Tips Penting:
1. Gunakan Scaffold sebagai struktur utama aplikasi
2. Pilih jenis TopAppBar yang sesuai dengan kebutuhan konten
3. Terapkan scroll behavior untuk pengalaman pengguna yang lebih baik
4. Gunakan FAB untuk aksi utama saja
5. Integrasikan semua komponen secara konsisten
6. Gunakan snackbar untuk feedback yang tidak mengganggu
7. Terapkan search inline untuk pengalaman pencarian yang mulus
8. Gunakan content padding untuk mencegah overlap dengan komponen UI
9. Kelola state secara efektif dalam struktur Scaffold
10. Gunakan Material3 components untuk konsistensi tampilan

Kamu sekarang siap membuat antarmuka aplikasi yang profesional dan konsisten dengan struktur Scaffold & AppBar!
