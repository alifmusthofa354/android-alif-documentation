### Basic TextField Implementation
**ui/screens/TextFieldExamplesScreen.kt**
```kotlin
package com.example.mycomposeapp.ui.screens

import androidx.compose.foundation.layout.*
import androidx.compose.foundation.rememberScrollState
import androidx.compose.foundation.text.KeyboardActions
import androidx.compose.foundation.text.KeyboardOptions
import androidx.compose.foundation.verticalScroll
import androidx.compose.material.icons.Icons
import androidx.compose.material.icons.filled.*
import androidx.compose.material3.*
import androidx.compose.runtime.*
import androidx.compose.ui.Alignment
import androidx.compose.ui.Modifier
import androidx.compose.ui.graphics.Color
import androidx.compose.ui.text.font.FontWeight
import androidx.compose.ui.text.input.*
import androidx.compose.ui.unit.dp
import androidx.compose.ui.unit.sp

@Composable
fun TextFieldExamplesScreen() {
    Column(
        modifier = Modifier
            .fillMaxSize()
            .verticalScroll(rememberScrollState())
            .padding(16.dp),
        verticalArrangement = Arrangement.spacedBy(16.dp)
    ) {
        Text("1Ô∏è‚É£ TextField & OutlinedTextField", fontSize = 20.sp, fontWeight = FontWeight.Bold)
        
        // 1. Basic TextField
        var basicText by remember { mutableStateOf("") }
        
        OutlinedTextField(
            value = basicText,
            onValueChange = { basicText = it },
            label = { Text("Basic TextField") },
            modifier = Modifier.fillMaxWidth()
        )
        
        // 2. TextField dengan Placeholder
        var placeholderText by remember { mutableStateOf("") }
        
        OutlinedTextField(
            value = placeholderText,
            onValueChange = { placeholderText = it },
            label = { Text("Nama Lengkap") },
            placeholder = { Text("Masukkan nama Anda...") },
            modifier = Modifier.fillMaxWidth()
        )
        
        // 3. TextField dengan Leading Icon
        var emailText by remember { mutableStateOf("") }
        
        OutlinedTextField(
            value = emailText,
            onValueChange = { emailText = it },
            label = { Text("Email") },
            leadingIcon = { Icon(Icons.Default.Email, contentDescription = null) },
            modifier = Modifier.fillMaxWidth()
        )
        
        // 4. TextField dengan Trailing Icon
        var passwordText by remember { mutableStateOf("") }
        var passwordVisible by remember { mutableStateOf(false) }
        
        OutlinedTextField(
            value = passwordText,
            onValueChange = { passwordText = it },
            label = { Text("Password") },
            trailingIcon = {
                IconButton(
                    onClick = { passwordVisible = !passwordVisible }
                ) {
                    Icon(
                        imageVector = if (passwordVisible) Icons.Default.Visibility else Icons.Default.VisibilityOff,
                        contentDescription = "Toggle password visibility"
                    )
                }
            },
            visualTransformation = if (passwordVisible) {
                VisualTransformation.None
            } else {
                PasswordVisualTransformation()
            },
            modifier = Modifier.fillMaxWidth()
        )
        
        // 5. Supporting Text
        var descriptionText by remember { mutableStateOf("") }
        
        OutlinedTextField(
            value = descriptionText,
            onValueChange = { descriptionText = it },
            label = { Text("Deskripsi") },
            supportingText = { Text("Deskripsi singkat tentang diri Anda") },
            modifier = Modifier.fillMaxWidth()
        )
        
        // 6. Error State
        var errorText by remember { mutableStateOf("") }
        val hasError = errorText.length < 3 && errorText.isNotBlank()
        
        OutlinedTextField(
            value = errorText,
            onValueChange = { errorText = it },
            label = { Text("Username") },
            supportingText = {
                if (hasError) {
                    Text("Username minimal 3 karakter", color = MaterialTheme.colorScheme.error)
                } else {
                    Text("Username akan digunakan untuk login")
                }
            },
            isError = hasError,
            trailingIcon = {
                if (hasError) {
                    Icon(Icons.Default.Error, contentDescription = "Error", tint = MaterialTheme.colorScheme.error)
                }
            },
            modifier = Modifier.fillMaxWidth()
        )
        
        // 7. Single Line vs Max Lines
        var singleLineText by remember { mutableStateOf("") }
        var multiLineText by remember { mutableStateOf("") }
        
        OutlinedTextField(
            value = singleLineText,
            onValueChange = { singleLineText = it },
            label = { Text("Single Line (Tidak bisa enter)") },
            singleLine = true,
            modifier = Modifier.fillMaxWidth()
        )
        
        OutlinedTextField(
            value = multiLineText,
            onValueChange = { multiLineText = it },
            label = { Text("Multi Line (Max 5 baris)") },
            maxLines = 5,
            modifier = Modifier
                .fillMaxWidth()
                .height(120.dp)
        )
        
        // 8. Read Only & Disabled
        var readOnlyText by remember { mutableStateOf("Ini hanya baca") }
        var disabledText by remember { mutableStateOf("Ini dinonaktifkan") }
        
        OutlinedTextField(
            value = readOnlyText,
            onValueChange = { readOnlyText = it },
            label = { Text("Read Only") },
            readOnly = true,
            modifier = Modifier.fillMaxWidth()
        )
        
        OutlinedTextField(
            value = disabledText,
            onValueChange = { disabledText = it },
            label = { Text("Disabled") },
            enabled = false,
            modifier = Modifier.fillMaxWidth()
        )
        
        // 9. TextField dengan Warna Kustom
        var customText by remember { mutableStateOf("") }
        
        OutlinedTextField(
            value = customText,
            onValueChange = { customText = it },
            label = { Text("Custom Colors") },
            colors = OutlinedTextFieldDefaults.colors(
                focusedBorderColor = MaterialTheme.colorScheme.primary,
                unfocusedBorderColor = MaterialTheme.colorScheme.secondary,
                focusedLabelColor = MaterialTheme.colorScheme.primary,
                unfocusedLabelColor = MaterialTheme.colorScheme.onSurfaceVariant
            ),
            modifier = Modifier.fillMaxWidth()
        )
    }
}
```

### üéì Penjelasan Konsep:
**TextField seperti Kotak Surat:**
- `value`: Isi dari TextField
- `onValueChange`: Callback saat teks berubah
- `label`: Teks label yang melayang
- `placeholder`: Teks bantuan saat kosong
- `leadingIcon`: Ikon di awal
- `trailingIcon`: Ikon di akhir
- `supportingText`: Teks bantuan di bawah
- `isError`: Status error
- `singleLine`: Hanya satu baris
- `maxLines`: Maksimal baris
- `readOnly`: Hanya bisa dibaca
- `enabled`: Aktif/tidak

@Composable
fun FormValidationExample() {
    var email by remember { mutableStateOf("") }
    var password by remember { mutableStateOf("") }
    
    val isEmailValid = remember(email) {
        email.contains("@") && email.contains(".")
    }
    
    val isPasswordValid = remember(password) {
        password.length >= 6
    }
    
    val isFormValid = remember(isEmailValid, isPasswordValid) {
        isEmailValid && isPasswordValid
    }
    
    Card(
        modifier = Modifier.fillMaxWidth()
    ) {
        Column(
            modifier = Modifier
                .fillMaxWidth()
                .padding(16.dp),
            verticalArrangement = Arrangement.spacedBy(12.dp)
        ) {
            Text("Form Validation", fontSize = 18.sp, fontWeight = FontWeight.Bold)
            
            OutlinedTextField(
                value = email,
                onValueChange = { email = it },
                label = { Text("Email") },
                leadingIcon = { Icon(Icons.Default.Email, contentDescription = null) },
                trailingIcon = {
                    if (email.isNotBlank()) {
                        Icon(
                            imageVector = if (isEmailValid) Icons.Default.Check else Icons.Default.Close,
                            contentDescription = null,
                            tint = if (isEmailValid) androidx.compose.ui.graphics.Color.Green else androidx.compose.ui.graphics.Color.Red
                        )
                    }
                },
                isError = email.isNotBlank() && !isEmailValid,
                supportingText = {
                    if (email.isNotBlank()) {
                        Text(
                            if (isEmailValid) "Email valid" else "Email tidak valid",
                            color = if (isEmailValid) androidx.compose.ui.graphics.Color.Green else androidx.compose.ui.graphics.Color.Red
                        )
                    }
                },
                modifier = Modifier.fillMaxWidth()
            )
            
            OutlinedTextField(
                value = password,
                onValueChange = { password = it },
                label = { Text("Password") },
                leadingIcon = { Icon(Icons.Default.Lock, contentDescription = null) },
                trailingIcon = {
                    if (password.isNotBlank()) {
                        Icon(
                            imageVector = if (isPasswordValid) Icons.Default.Check else Icons.Default.Close,
                            contentDescription = null,
                            tint = if (isPasswordValid) androidx.compose.ui.graphics.Color.Green else androidx.compose.ui.graphics.Color.Red
                        )
                    }
                },
                isError = password.isNotBlank() && !isPasswordValid,
                supportingText = {
                    if (password.isNotBlank()) {
                        Text(
                            if (isPasswordValid) "Password valid" else "Minimal 6 karakter",
                            color = if (isPasswordValid) androidx.compose.ui.graphics.Color.Green else androidx.compose.ui.graphics.Color.Red
                        )
                    }
                },
                visualTransformation = PasswordVisualTransformation(),
                modifier = Modifier.fillMaxWidth()
            )
            
            Button(
                onClick = { /* Login */ },
                enabled = isFormValid,
                modifier = Modifier.fillMaxWidth()
            ) {
                Text("Login")
            }
            
            if (isFormValid) {
                Text("Form valid - siap untuk submit!", color = androidx.compose.ui.graphics.Color.Green)
            }
        }
    }
}

@Composable
fun ErrorHandlingExample() {
    var username by remember { mutableStateOf("") }
    var error by remember { mutableStateOf<String?>(null) }
    
    val validateUsername = { input: String ->
        if (input.length < 3) {
            "Username minimal 3 karakter"
        } else if (input.length > 20) {
            "Username maksimal 20 karakter"
        } else if (!input.matches(Regex("^[a-zA-Z0-9_]*$"))) {
            "Username hanya boleh huruf, angka, dan underscore"
        } else {
            null
        }
    }
    
    Card(
        modifier = Modifier.fillMaxWidth()
    ) {
        Column(
            modifier = Modifier
                .fillMaxWidth()
                .padding(16.dp),
            verticalArrangement = Arrangement.spacedBy(12.dp)
        ) {
            Text("Error Handling", fontSize = 18.sp, fontWeight = FontWeight.Bold)
            
            OutlinedTextField(
                value = username,
                onValueChange = { 
                    username = it
                    error = validateUsername(it)
                },
                label = { Text("Username") },
                leadingIcon = { Icon(Icons.Default.Person, contentDescription = null) },
                trailingIcon = {
                    if (username.isNotEmpty()) {
                        IconButton(
                            onClick = { username = "" }
                        ) {

                    }

fun TriStateCheckboxExample() {
    var allChecked by remember { mutableStateOf(false) }
    var individualOptions by remember {
        mutableStateOf(
            mutableListOf(
                "Opsi 1" to false,
                "Opsi 2" to false,
                "Opsi 3" to false
            )
        )
    }
    
    // Hitung status intermediate
    val checkedCount = individualOptions.count { it.second }
    val isIndeterminate = checkedCount > 0 && checkedCount < individualOptions.size
    
    Card(
        modifier = Modifier.fillMaxWidth()
    ) {
        Column(
            modifier = Modifier
                .fillMaxWidth()
                .padding(16.dp)
        ) {
            Text("Tri-state Checkbox", fontWeight = FontWeight.Bold)
            
            // Master checkbox
            Row(
                verticalAlignment = Alignment.CenterVertically,
                horizontalArrangement = Arrangement.spacedBy(8.dp)
            ) {
                Checkbox(
                    checked = allChecked,
                    onCheckedChange = { checked ->
                        allChecked = checked
                        individualOptions = individualOptions.map { 
                            Pair(it.first, checked) 
                        }.toMutableList()
                    },
                    // TriState untuk indeterminate state
                    interactionSource = remember { MutableInteractionSource() }
                )
                Text("Pilih Semua")
            }
            
            Spacer(modifier = Modifier.height(8.dp))
            
            // Individual checkboxes
            individualOptions.forEachIndexed { index, (name, checked) ->
                Row(
                    modifier = Modifier.fillMaxWidth(),
                    verticalAlignment = Alignment.CenterVertically,
                    horizontalArrangement = Arrangement.spacedBy(8.dp)
                ) {
                    Checkbox(
                        checked = checked,
                        onCheckedChange = { newChecked ->
                            individualOptions[index] = Pair(name, newChecked)
                            
                            // Update master checkbox state
                            val allSelected = individualOptions.all { it.second }
                            val noneSelected = individualOptions.none { it.second }
                            
                            if (allSelected) {
                                allChecked = true
                            } else if (noneSelected) {
                                allChecked = false
                            } else {
                                allChecked = true // Ini akan membuat tri-state
                            }
                        }
                    )
                    Text(name)
                }
            }
            
            Text(
                text = "Status: ${
                    when {
                        allChecked -> "Semua dipilih"
                        checkedCount == 0 -> "Tidak ada yang dipilih"
                        else -> "Sebagian dipilih ($checkedCount/${individualOptions.size})"
                    }
                }",
                fontSize = 12.sp,
                color = MaterialTheme.colorScheme.onSurfaceVariant
            )
        }
