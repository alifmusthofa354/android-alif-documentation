## 1Ô∏è‚É£ AlertDialog üí¨

### Konsep Dasar
AlertDialog seperti peringatan penting üö® - muncul di tengah layar untuk menginformasikan atau memberikan pilihan penting!

### Implementation
**ui/screens/AlertDialogExamplesScreen.kt**
```kotlin
package com.example.mycomposeapp.ui.screens

import androidx.compose.foundation.layout.*
import androidx.compose.material.icons.Icons
import androidx.compose.material.icons.filled.*
import androidx.compose.material3.*
import androidx.compose.runtime.*
import androidx.compose.ui.Alignment
import androidx.compose.ui.Modifier
import androidx.compose.ui.text.font.FontWeight
import androidx.compose.ui.unit.dp
import androidx.compose.ui.unit.sp

@Composable
fun AlertDialogExamplesScreen() {
    Column(
        modifier = Modifier
            .fillMaxSize()
            .padding(16.dp),
        verticalArrangement = Arrangement.spacedBy(16.dp)
    ) {
        Text("1Ô∏è‚É£ AlertDialog", fontSize = 20.sp, fontWeight = FontWeight.Bold)
        
        // 1. Basic AlertDialog
        var showDialog by remember { mutableStateOf(false) }
        
        Button(
            onClick = { showDialog = true },
            modifier = Modifier.fillMaxWidth()
        ) {
            Text("Tampilkan Basic AlertDialog")
        }
        
        if (showDialog) {
            AlertDialog(
                onDismissRequest = { showDialog = false },
                title = { Text("Konfirmasi") },
                text = { Text("Apakah Anda yakin ingin melanjutkan?") },
                confirmButton = {
                    Button(
                        onClick = {
                            // Handle confirm
                            showDialog = false
                        }
                    ) {
                        Text("Ya")
                    }
                },
                dismissButton = {
                    OutlinedButton(
                        onClick = { showDialog = false }
                    ) {
                        Text("Tidak")
                    }
                }
            )
        }
        
        // 2. AlertDialog with Icon
        var showIconDialog by remember { mutableStateOf(false) }
        
        Button(
            onClick = { showIconDialog = true },
            modifier = Modifier.fillMaxWidth()
        ) {
            Text("Tampilkan AlertDialog dengan Icon")
        }
        
        if (showIconDialog) {
            AlertDialog(
                onDismissRequest = { showIconDialog = false },
                icon = { Icon(Icons.Default.Warning, contentDescription = null) },
                title = { Text("Peringatan") },
                text = { Text("Tindakan ini tidak bisa dibatalkan") },
                confirmButton = {
                    Button(
                        onClick = { showIconDialog = false }
                    ) {
                        Text("Saya Mengerti")
                    }
                },
                dismissButton = {
                    OutlinedButton(
                        onClick = { showIconDialog = false }
                    ) {
                        Text("Batal")
                    }
                }
            )
        }
        
        // 3. AlertDialog without dismiss button
        var showConfirmDialog by remember { mutableStateOf(false) }
        
        Button(
            onClick = { showConfirmDialog = true },
            modifier = Modifier.fillMaxWidth()
        ) {
            Text("Tampilkan AlertDialog Konfirmasi")
        }
        
        if (showConfirmDialog) {
            AlertDialog(
                onDismissRequest = { /* Tidak bisa ditutup dengan tap di luar */ },
                title = { Text("Penghapusan Data") },
                text = { 
                    Text("Semua data akan dihapus secara permanen. Tindakan ini tidak bisa dibatalkan.") 
                },
                confirmButton = {
                    Button(
                        onClick = { 
                            // Handle deletion
                            showConfirmDialog = false
                        },
                        colors = ButtonDefaults.buttonColors(
                            containerColor = MaterialTheme.colorScheme.error
                        )
                    ) {
                        Text("Hapus Semua")
                    }
                },
                dismissButton = {
                    OutlinedButton(
                        onClick = { showConfirmDialog = false }
                    ) {
                        Text("Batal")
                    }
                }
            )
        }
        
        // 4. Custom content in dialog
        var showCustomDialog by remember { mutableStateOf(false) }
        
        Button(
            onClick = { showCustomDialog = true },
            modifier = Modifier.fillMaxWidth()
        ) {
            Text("Tampilkan AlertDialog Custom")
        }
        
        if (showCustomDialog) {
            var selectedOption by remember { mutableStateOf(0) }
            val options = listOf("Opsi 1", "Opsi 2", "Opsi 3")
            
            AlertDialog(
                onDismissRequest = { showCustomDialog = false },
                title = { Text("Pilih Opsi") },
                text = {
                    Column {
                        options.forEachIndexed { index, option ->
                            Row(
                                modifier = Modifier
                                    .fillMaxWidth()
                                    .clickable { selectedOption = index }
                                    .padding(8.dp),
                                verticalAlignment = Alignment.CenterVertically
                            ) {
                                RadioButton(
                                    selected = selectedOption == index,
                                    onClick = { selectedOption = index }
                                )
                                Text(option)
                            }
                        }
                    }
                },
                confirmButton = {
                    Button(
                        onClick = {
                            // Handle selection
                            showCustomDialog = false
                        }
                    ) {
                        Text("Pilih")
                    }
                },
                dismissButton = {
                    OutlinedButton(
                        onClick = { showCustomDialog = false }
                    ) {
                        Text("Batal")
                    }
                }
            )
        }
        
        // 5. Error dialog
        var showErrorDialog by remember { mutableStateOf(false) }
        
        Button(
            onClick = { showErrorDialog = true },
            modifier = Modifier.fillMaxWidth()
        ) {
            Text("Tampilkan Error Dialog")
        }
        
        if (showErrorDialog) {
            AlertDialog(
                onDismissRequest = { showErrorDialog = false },
                icon = { Icon(Icons.Default.Error, contentDescription = null, tint = MaterialTheme.colorScheme.error) },
                title = { Text("Terjadi Kesalahan", color = MaterialTheme.colorScheme.error) },
                text = { Text("Tidak dapat menyambung ke server. Silakan periksa koneksi internet Anda.") },
                confirmButton = {
                    Button(
                        onClick = { showErrorDialog = false }
                    ) {
                        Text("Coba Lagi")
                    }
                },
                containerColor = MaterialTheme.colorScheme.surface,
                titleContentColor = MaterialTheme.colorScheme.error
            )
        }
    }
}
```

### üéì Penjelasan Konsep:
**AlertDialog seperti Peringatan Penting:**
- `onDismissRequest`: Callback saat dialog ditutup
- `title`: Judul dialog
- `text`: Isi utama dialog
- `confirmButton`: Tombol konfirmasi
- `dismissButton`: Tombol batal
- `icon`: Ikon di atas judul

**Jenis AlertDialog:**
- Basic: Dialog dengan title dan text sederhana
- With Icon: Dialog dengan ikon untuk visual yang lebih kuat
- Custom Content: Dialog dengan konten kustom (radio button, checkbox, dll)

## 3Ô∏è‚É£ Snackbar üîî

### Konsep Dasar

### Implementation
**ui/screens/SnackbarExamplesScreen.kt**
```kotlin
package com.example.mycomposeapp.ui.screens

import androidx.compose.foundation.layout.*
import androidx.compose.material.icons.Icons
import androidx.compose.material.icons.filled.*
import androidx.compose.material3.*
import androidx.compose.runtime.*
import androidx.compose.ui.Alignment
import androidx.compose.ui.Modifier
import androidx.compose.ui.text.font.FontWeight
import androidx.compose.ui.unit.dp
import androidx.compose.ui.unit.sp
import kotlinx.coroutines.launch

@Composable
fun SnackbarExamplesScreen() {
    val snackbarHostState = remember { SnackbarHostState() }
    val coroutineScope = rememberCoroutineScope()
    
    Box(
        modifier = Modifier.fillMaxSize()
    ) {
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(16.dp),
            verticalArrangement = Arrangement.spacedBy(16.dp)
        ) {
            Text("3Ô∏è‚É£ Snackbar", fontSize = 20.sp, fontWeight = FontWeight.Bold)
            
            // 1. Basic Snackbar
            Button(
                onClick = {
                    coroutineScope.launch {
                        snackbarHostState.showSnackbar("Ini adalah snackbar dasar!")
                    }
                },
                modifier = Modifier.fillMaxWidth()
            ) {
                Text("Tampilkan Basic Snackbar")
            }
            
            // 2. Snackbar with Action Button
            Button(
                onClick = {
                    coroutineScope.launch {
                        val result = snackbarHostState.showSnackbar(
                            message = "Item berhasil dihapus",
                            actionLabel = "Undo"
                        )
                        // Handle action result
                        if (result == SnackbarResult.ActionPerformed) {
                            // Undo action
                            println("Undo action performed")
                        }
                    }
                },
                modifier = Modifier.fillMaxWidth()
            ) {
                Text("Tampilkan Snackbar dengan Aksi")
            }
            
            // 3. Snackbar with Different Durations
            Row(
                modifier = Modifier.fillMaxWidth(),
                horizontalArrangement = Arrangement.spacedBy(8.dp)
            ) {
                Button(
                    onClick = {
                        coroutineScope.launch {
                            snackbarHostState.showSnackbar(
                                message = "Durasi SHORT",
                                duration = SnackbarDuration.Short
                            )
                        }
                    },
                    modifier = Modifier.weight(1f)
                ) {
                    Text("SHORT")
                }
                
                Button(
                    onClick = {
                        coroutineScope.launch {
                            snackbarHostState.showSnackbar(
                                message = "Durasi LONG",
                                duration = SnackbarDuration.Long
                            )
                        }
                    },
                    modifier = Modifier.weight(1f)
                ) {
                    Text("LONG")
                }
                
                Button(
                    onClick = {
                        coroutineScope.launch {
                            snackbarHostState.showSnackbar(
                                message = "Durasi INDEFINITE",
                                duration = SnackbarDuration.Indefinite
                            )
                        }
                    },
                    modifier = Modifier.weight(1f)
                ) {
                    Text("INDEFINITE")
                }
            }
            
            // 4. Custom Snackbar
            Button(
                onClick = {
                    coroutineScope.launch {
                        snackbarHostState.showSnackbar(
                            message = "Operasi berhasil",
                            actionLabel = "Lihat",
                            duration = SnackbarDuration.Short
                        )
                    }
                },
                modifier = Modifier.fillMaxWidth()
            ) {
                Text("Tampilkan Custom Snackbar")
            }
            
            // 5. Error Snackbar
            Button(
                onClick = {
                    coroutineScope.launch {
                        val result = snackbarHostState.showSnackbar(
                            message = "Terjadi kesalahan jaringan",
                            actionLabel = "Coba Lagi",
                            duration = SnackbarDuration.Indefinite
                        )
                        if (result == SnackbarResult.ActionPerformed) {
                            // Handle retry
                            println("Retry action performed")
                        }
                    }
                },
                modifier = Modifier.fillMaxWidth()
            ) {
                Text("Tampilkan Error Snackbar")
            }
            
            // 6. Success Snackbar
            Button(
                onClick = {
                    coroutineScope.launch {
                        snackbarHostState.showSnackbar(
                            message = "Data berhasil disimpan",
                            actionLabel = "Lihat",
                            duration = SnackbarDuration.Short
                        )
                    }
                },
                modifier = Modifier.fillMaxWidth()
            ) {
                Text("Tampilkan Success Snackbar")
            }
        }
        
        // SnackbarHost harus ditempatkan di sini untuk menampilkan snackbar
        SnackbarHost(
            hostState = snackbarHostState,
            modifier = Modifier
                .align(Alignment.BottomCenter)
                .padding(16.dp)
        )
    }
}
```
